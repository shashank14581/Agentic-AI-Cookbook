Pattern 14 · Vector Search

Agents who Search Vectors

In this pattern, we move from chunk-level retrieval to a clean vector search index. The agent stores small knowledge documents as embeddings, retrieves the most similar documents for a user question, and answers only from those retrieved documents.

OVERVIEW

What this pattern teaches

The Vector Search Agent pattern shows how an agent can search across a small private knowledge base before generating an answer. Each knowledge item becomes a vector. The question also becomes a vector. The agent compares them using cosine similarity and passes only the top matching documents to Gemini.

This pattern is sequenced after Pattern 13. Pattern 13 teaches chunking and embedding from a single document. Pattern 14 teaches a cleaner retrieval index over multiple knowledge records. This is the bridge from toy RAG memory to a search-backed agent architecture.

import os
import torch
import torch.nn as nn
import torch.nn.functional as F
from google import genai
from google.genai import types

os.environ["GEMINI_API_KEY"] = "your_api_key"


class SimpleEmbeddingModel(nn.Module):
    def __init__(self, vocab_size, embedding_dim=32):
        super().__init__()
        self.embedding = nn.Embedding(vocab_size, embedding_dim)

    def forward(self, token_ids):
        word_embeddings = self.embedding(token_ids)
        return word_embeddings.mean(dim=0)


class VectorSearchIndex:
    def __init__(self):
        self.documents = []

    def add(self, doc_id, text, embedding):
        self.documents.append({
            "doc_id": doc_id,
            "text": text,
            "embedding": embedding
        })

    def search(self, query_embedding, top_k=3):
        results = []

        for doc in self.documents:
            similarity = F.cosine_similarity(
                query_embedding.unsqueeze(0),
                doc["embedding"].unsqueeze(0)
            ).item()

            results.append((similarity, doc))

        results = sorted(results, key=lambda x: x[0], reverse=True)
        return results[:top_k]


class VectorSearchMemory:
    def __init__(self, embedding_dim=32):
        self.word_to_id = {"<UNK>": 0}
        self.embedding_dim = embedding_dim
        self.model = None
        self.index = VectorSearchIndex()

    def build_vocab(self, texts):
        for text in texts:
            for word in text.lower().split():
                if word not in self.word_to_id:
                    self.word_to_id[word] = len(self.word_to_id)

    def encode_text(self, text):
        token_ids = [
            self.word_to_id.get(word, 0)
            for word in text.lower().split()
        ]

        return torch.tensor(token_ids, dtype=torch.long)

    def build_index(self, texts):
        self.build_vocab(texts)

        self.model = SimpleEmbeddingModel(
            vocab_size=len(self.word_to_id),
            embedding_dim=self.embedding_dim
        )

        for i, text in enumerate(texts, start=1):
            token_ids = self.encode_text(text)

            with torch.no_grad():
                embedding = self.model(token_ids)

            self.index.add(
                doc_id=f"doc_{i}",
                text=text,
                embedding=embedding
            )

            print(f"Indexed doc_{i}")

    def retrieve(self, query, top_k=3):
        token_ids = self.encode_text(query)

        with torch.no_grad():
            query_embedding = self.model(token_ids)

        results = self.index.search(query_embedding, top_k=top_k)

        context = ""

        for score, doc in results:
            context += f"""
Document: {doc["doc_id"]}
Similarity: {round(score, 4)}
Text: {doc["text"]}
"""

        return context


class BaseAgent:
    def __init__(self, name, sys_prompt, vector_memory, model="gemini-flash-latest"):
        self.name = name
        self.sys_prompt = sys_prompt
        self.vector_memory = vector_memory
        self.client = genai.Client(api_key=os.environ.get("GEMINI_API_KEY"))
        self.model = model
        self.memory = []

    def think(self, question):
        retrieved_context = self.vector_memory.retrieve(
            query=question,
            top_k=3
        )

        final_input = f"""
You are using Vector Search before generation.

Step 1: Retrieve relevant documents.
Step 2: Use only retrieved documents to answer.

Retrieved Documents:
{retrieved_context}

Question:
{question}

Answer only from the retrieved documents.
If the retrieved documents do not contain the answer, say:
"I do not have enough retrieved context to answer."
"""

        contents = [
            types.Content(
                role="user",
                parts=[types.Part.from_text(text=final_input)]
            )
        ]

        config = types.GenerateContentConfig(
            thinking_config=types.ThinkingConfig(thinking_budget=0),
            system_instruction=[types.Part.from_text(text=self.sys_prompt)]
        )

        response = ""

        for chunk in self.client.models.generate_content_stream(
            model=self.model,
            contents=contents,
            config=config
        ):
            if chunk.text:
                print(chunk.text, end="")
                response += chunk.text

        print()
        self.memory.append((question, response))
        return response


class VectorSearchAgent(BaseAgent):
    def __init__(self, name, vector_memory):
        super().__init__(
            name,
            """
You are a Vector Search RAG Agent.

You always retrieve before generating.
You answer only from retrieved context.
You do not guess.
""",
            vector_memory
        )


def orchestrate(agent, questions):
    for question in questions:
        print(f"\nUser: {question}")
        print(f"{agent.name}: ", end="")
        agent.think(question)

Mental Model

Lines 1-8: The script imports PyTorch, Gemini SDK objects, and sets a placeholder API key. In a real repo, the key should come from the environment, not from committed source code.

Lines 11-18: SimpleEmbeddingModel is the same teaching model from the previous pattern. It converts token IDs into vectors and averages them into one document vector.

Lines 21-44: VectorSearchIndex is the search layer. It stores document IDs, text, and embeddings, then ranks documents using cosine similarity.

Lines 47-112: VectorSearchMemory builds the vocabulary, encodes text into IDs, creates embeddings for each knowledge record, and retrieves the top matching records for a query.

Lines 115-181: BaseAgent performs the agent loop: retrieve first, build a constrained prompt, stream the Gemini answer, then save the response in memory.

Lines 184-198: VectorSearchAgent gives the base agent a strict identity: retrieve before generating, answer only from context, and do not guess.

Lines 201-238: The orchestration section indexes the private knowledge base, creates the agent, asks test questions, and prints the grounded answers. The missing commas from the original script are fixed here so the list items do not accidentally merge.
