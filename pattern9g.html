Pattern 9g · Agentic RAG

Agentic RAG Router

This pattern upgrades basic RAG with a routing decision. The agent first asks another lightweight agent whether retrieval is needed. Private or operational questions trigger RAG; general questions skip retrieval and are answered normally.

Decide before retrieving.

The agent learns a simple discipline: not every question needs private context, but private questions must be grounded.

OVERVIEW

What this pattern teaches

Pattern 9g introduces agentic control into the RAG loop. Pattern 9f always retrieved before answering. Pattern 9g adds a RetrievalDecisionAgent that decides whether a question needs private knowledge retrieval.

This is the first important step from a fixed RAG pipeline toward an agentic system. The flow becomes conditional: question → routing decision → retrieve or skip → generate. That makes the agent more flexible while still protecting private project facts from hallucination.

import os
import torch
import torch.nn as nn
import torch.nn.functional as F
from google import genai
from google.genai import types

os.environ["GEMINI_API_KEY"] = "your_api_key"


class SimpleEmbeddingModel(nn.Module):
    def __init__(self, vocab_size, embedding_dim=32):
        super().__init__()
        self.embedding = nn.Embedding(vocab_size, embedding_dim)

    def forward(self, token_ids):
        word_embeddings = self.embedding(token_ids)
        return word_embeddings.mean(dim=0)


class RAGMemory:
    def __init__(self, embedding_dim=32):
        self.word_to_id = {"<UNK>": 0}
        self.embedding_dim = embedding_dim
        self.model = None
        self.store = []

    def build_vocab(self, texts):
        for text in texts:
            for word in text.lower().split():
                if word not in self.word_to_id:
                    self.word_to_id[word] = len(self.word_to_id)

    def encode_text(self, text):
        return torch.tensor([
            self.word_to_id.get(w, 0)
            for w in text.lower().split()
        ], dtype=torch.long)

    def build_index(self, knowledge_base):
        self.build_vocab(knowledge_base)
        self.model = SimpleEmbeddingModel(
            len(self.word_to_id),
            self.embedding_dim
        )

        for i, text in enumerate(knowledge_base, start=1):
            token_ids = self.encode_text(text)

            with torch.no_grad():
                embedding = self.model(token_ids)

            self.store.append({
                "doc_id": f"doc_{i}",
                "text": text,
                "embedding": embedding
            })

            print(f"Indexed doc_{i}")

    def retrieve(self, query, top_k=3):
        query_ids = self.encode_text(query)

        with torch.no_grad():
            query_embedding = self.model(query_ids)

        results = []

        for item in self.store:
            score = F.cosine_similarity(
                query_embedding.unsqueeze(0),
                item["embedding"].unsqueeze(0)
            ).item()

            results.append((score, item))

        results = sorted(results, key=lambda x: x[0], reverse=True)
        context = ""

        for score, item in results[:top_k]:
            context += f"""
Document: {item["doc_id"]}
Similarity: {round(score, 4)}
Text: {item["text"]}
"""

        return context


class BaseAgent:
    def __init__(self, name, sys_prompt, model="gemini-flash-latest"):
        self.name = name
        self.sys_prompt = sys_prompt
        self.memory = []
        self.client = genai.Client(api_key=os.environ.get("GEMINI_API_KEY"))
        self.model = model

    def think(self, input_text, print_output=True):
        contents = [
            types.Content(
                role="user",
                parts=[types.Part.from_text(text=input_text)]
            )
        ]

        config = types.GenerateContentConfig(
            thinking_config=types.ThinkingConfig(thinking_budget=0),
            system_instruction=[types.Part.from_text(text=self.sys_prompt)]
        )

        response = ""

        for chunk in self.client.models.generate_content_stream(
            model=self.model,
            contents=contents,
            config=config
        ):
            if chunk.text:
                if print_output:
                    print(chunk.text, end="")
                response += chunk.text

        if print_output:
            print()

        self.memory.append((input_text, response))
        return response


class RetrievalDecisionAgent(BaseAgent):
    def __init__(self):
        super().__init__(
            name="RetrievalDecisionAgent",
            sys_prompt="""
You are a retrieval decision agent.

Your job is to decide whether a question needs private knowledge retrieval.

Return only one word:
RETRIEVE
or
NO_RETRIEVE

Use RETRIEVE when the question asks about private projects, internal terms, unknown codenames, owners, thresholds, queues, tables, rules, or operational details.

Use NO_RETRIEVE when the question is general knowledge or simple conversation.
"""
        )

    def should_retrieve(self, question):
        decision = self.think(question, print_output=False)
        decision = decision.strip().upper()
        return "RETRIEVE" in decision


class AgenticRAGAgent(BaseAgent):
    def __init__(self, name, rag_memory):
        super().__init__(
            name=name,
            sys_prompt="""
You are an Agentic RAG Agent.

You decide whether retrieval is needed.
If retrieved context is provided, answer only from that context.
If no retrieved context is provided, answer normally using general reasoning.
Do not invent private project facts.
"""
        )
        self.rag_memory = rag_memory
        self.router = RetrievalDecisionAgent()

    def think(self, question):
        needs_retrieval = self.router.should_retrieve(question)

        if needs_retrieval:
            print("[Decision: RETRIEVE]")
            retrieved_context = self.rag_memory.retrieve(
                query=question,
                top_k=3
            )

            final_input = f"""
The agent decided retrieval is needed.

Retrieved Context:
{retrieved_context}

Question:
{question}

Rules:
- Answer only from retrieved context.
- Do not use outside knowledge for private facts.
- If context is insufficient, say: "I do not have enough retrieved context to answer."
"""
        else:
            print("[Decision: NO_RETRIEVE]")
            final_input = f"""
The agent decided retrieval is not needed.

Question:
{question}

Answer normally.
"""

        return super().think(final_input)


def orchestrate(agent, questions):
    for question in questions:
        print(f"\nUser: {question}")
        print(f"{agent.name}: ", end="")
        agent.think(question)

Mental Model

Lines 1-8: Import PyTorch, Gemini SDK components, and set a placeholder API key. In production, keep the API key outside the source file.

Lines 11-18: SimpleEmbeddingModel creates average embeddings from token IDs. It is intentionally simple so the retrieval mechanics stay visible.

Lines 21-91: RAGMemory builds vocabulary, encodes text, embeds the private knowledge base, and retrieves the most similar records using cosine similarity.

Lines 94-136: BaseAgent is now more generic. It can stream a response for any input text and optionally suppress printing, which is useful for the router.

Lines 139-164: RetrievalDecisionAgent is the router. It returns either RETRIEVE or NO_RETRIEVE based on whether the question appears to require private project knowledge.

Lines 167-218: AgenticRAGAgent asks the router first. If retrieval is needed, it fetches context and forces grounded answering. If not, it allows normal reasoning.

Lines 221-256: The main block demonstrates both branches: general questions such as What is RAG? skip retrieval, while private Blue Mango questions trigger retrieval.
