What this pattern teaches

Pattern 21 is the first complete Agent + MCP integration. Pattern 20 created an MCP server. Pattern 19 created an MCP client. Pattern 21 combines both with Gemini function declarations so the model can choose a tool, pass arguments, and receive the result through the MCP layer.

The key upgrade is multiple tool availability. The agent can call search_private_knowledge for internal project facts or calculate_campaign_risk when the user provides a confidence score. This is where tool use begins to feel like real agentic behavior.

import os
import time
import socket
import threading
import nest_asyncio

from google import genai
from google.genai import types

from mcp.server.fastmcp import FastMCP
from mcp import ClientSession
from mcp.client.streamable_http import streamablehttp_client

nest_asyncio.apply()

os.environ["GEMINI_API_KEY"] = "your_api_key"


# ============================================================
# CONFIG
# ============================================================

PORT = 8022
MCP_URL = f"http://127.0.0.1:{PORT}/mcp"


# ============================================================
# MCP SERVER
# ============================================================

mcp = FastMCP(
    name="agent_tool_server",
    stateless_http=True,
    json_response=True
)


@mcp.tool()
def search_private_knowledge(query: str) -> str:
    query = query.lower()

    if "blue mango" in query and any(word in query for word in ["owner", "ownership", "failure", "failures"]):
        return "The fallback owner for Blue Mango failures is Aniket from the DS Ops team."

    if "blue mango" in query:
        return "Blue Mango is the internal codename for the customer retention model in Project Sarayu."

    if "confidence" in query or "0.62" in query or "threshold" in query:
        return "If Blue Mango confidence drops below 0.62, the audience must be routed to the Silver Gate manual review queue."

    if "silver gate" in query:
        return "Silver Gate is the manual review queue for Project Sarayu."

    return "No private knowledge found."


@mcp.tool()
def calculate_campaign_risk(confidence: float) -> str:
    if confidence < 0.62:
        return "HIGH RISK: Route audience to Silver Gate manual review queue."

    return "LOW RISK: Audience can continue to activation."


def run_mcp_server():
    mcp.settings.port = PORT
    mcp.run(transport="streamable-http")


def port_is_busy(port):
    with socket.socket(socket.AF_INET, socket.SOCK_STREAM) as s:
        return s.connect_ex(("127.0.0.1", port)) == 0


if not port_is_busy(PORT):
    server_thread = threading.Thread(
        target=run_mcp_server,
        daemon=True
    )
    server_thread.start()
    time.sleep(3)
    print(f"MCP server started on port {PORT}.")
else:
    print(f"MCP server already running on port {PORT}. Reusing it.")


# ============================================================
# MCP CLIENT
# ============================================================

class MCPClient:
    def __init__(self, url=MCP_URL):
        self.url = url

    async def call_tool(self, tool_name, arguments):
        async with streamablehttp_client(self.url) as (read, write, _):
            async with ClientSession(read, write) as session:
                await session.initialize()

                result = await session.call_tool(
                    name=tool_name,
                    arguments=arguments
                )

                return result.content[0].text


# ============================================================
# BASE AI AGENT
# ============================================================

class BaseAgent:
    def __init__(self, name, sys_prompt, mcp_client, model="gemini-2.5-flash-lite"):
        self.name = name
        self.sys_prompt = sys_prompt
        self.mcp_client = mcp_client
        self.client = genai.Client(api_key=os.environ.get("GEMINI_API_KEY"))
        self.model = model
        self.memory = []

    def get_tools(self):
        return [
            types.Tool(
                function_declarations=[
                    types.FunctionDeclaration(
                        name="search_private_knowledge",
                        description="Search private MCP knowledge base for internal project facts.",
                        parameters=types.Schema(
                            type=types.Type.OBJECT,
                            properties={
                                "query": types.Schema(
                                    type=types.Type.STRING,
                                    description="Search query for private knowledge"
                                )
                            },
                            required=["query"]
                        )
                    ),
                    types.FunctionDeclaration(
                        name="calculate_campaign_risk",
                        description="Calculate campaign risk based on Blue Mango confidence score.",
                        parameters=types.Schema(
                            type=types.Type.OBJECT,
                            properties={
                                "confidence": types.Schema(
                                    type=types.Type.NUMBER,
                                    description="Blue Mango confidence score"
                                )
                            },
                            required=["confidence"]
                        )
                    )
                ]
            )
        ]

    async def think(self, input_text):
        response = self.client.models.generate_content(
            model=self.model,
            contents=[
                types.Content(
                    role="user",
                    parts=[types.Part.from_text(text=input_text)]
                )
            ],
            config=types.GenerateContentConfig(
                thinking_config=types.ThinkingConfig(thinking_budget=0),
                system_instruction=[types.Part.from_text(self.sys_prompt)],
                tools=self.get_tools()
            )
        )

        part = response.candidates[0].content.parts[0]

        if part.function_call:
            tool_name = part.function_call.name
            arguments = dict(part.function_call.args)

            print(f"\n[Gemini selected MCP tool: {tool_name}]")
            print(arguments)

            answer = await self.mcp_client.call_tool(
                tool_name=tool_name,
                arguments=arguments
            )

        else:
            answer = response.text

        print(answer)
        self.memory.append((input_text, answer))
        return answer


# ============================================================
# AGENT + MCP INTEGRATION
# ============================================================

class MCPAgent(BaseAgent):
    def __init__(self, name, mcp_client):
        super().__init__(
            name=name,
            sys_prompt="""
You are an MCP-enabled AI agent.

You have access to tools through an MCP server.

Use search_private_knowledge when the user asks about:
- Blue Mango
- Silver Gate
- Project Sarayu
- internal owners
- private project facts

Use calculate_campaign_risk when the user provides a confidence score.

Do not invent internal facts.
""",
            mcp_client=mcp_client
        )


# ============================================================
# ORCHESTRATION
# ============================================================

async def main():
    mcp_client = MCPClient()

    agent = MCPAgent(
        name="MCPAgent",
        mcp_client=mcp_client
    )

    questions = [
        "What is Blue Mango?",
        "Who owns Blue Mango failures?",
        "What happens if confidence drops below 0.62?",
        "Blue Mango confidence is 0.55. What should we do?",
        "Blue Mango confidence is 0.81. What should we do?"
    ]

    for question in questions:
        print(f"\nUser: {question}")
        print(f"{agent.name}: ", end="")
        await agent.think(question)


await main()
Mental Model

Lines 1-18: Install and import everything needed for Gemini, MCP server, MCP client, threading, and notebook-safe async execution.

Lines 25-26: Configure the MCP endpoint on port 8022. The agent and client will call tools through this URL.

Lines 33-70: Create the MCP server and register two tools: search_private_knowledge for private facts and calculate_campaign_risk for confidence-score decisions.

Lines 73-93: Start the MCP server in a background daemon thread only if the port is free. This protects notebook reruns from duplicate server errors.

Lines 100-115: MCPClient connects to the server over streamable HTTP, initializes a session, calls the selected tool, and returns the result text.

Lines 122-232: BaseAgent declares both MCP tools to Gemini using function declarations. Gemini can choose a tool, provide arguments, and Python executes the tool through the MCP client.

Lines 239-264: MCPAgent gives the model tool-selection rules: use private search for internal facts, use campaign risk calculation for confidence scores, and do not invent internal facts.

Lines 271-294: The orchestration asks both private-knowledge questions and score-based action questions, showing that the same agent can route to different MCP tools.
