OVERVIEW

What this pattern teaches

Pattern 18 moves beyond in-memory RAG. Instead of embedding and searching inside Python lists, the agent exposes private knowledge as an MCP tool named search_private_knowledge. Gemini receives that tool schema, decides when to call it, and the Python agent executes the call through an MCP client.

The sequencing is important: first install dependencies, then start the MCP server, then create the MCP client, then declare the Gemini tool, then let the agent call the tool when private facts are needed. This makes the architecture feel closer to a real agent system than a simple RAG demo.

import os
import time
import socket
import threading
import nest_asyncio

from google import genai
from google.genai import types

from mcp.server.fastmcp import FastMCP
from mcp import ClientSession
from mcp.client.streamable_http import streamablehttp_client

nest_asyncio.apply()

os.environ["GEMINI_API_KEY"] = "your_api_key"


# ============================================================
# CONFIG
# ============================================================

PORT = 8011
MCP_URL = f"http://127.0.0.1:{PORT}/mcp"


# ============================================================
# REAL MCP SERVER
# ============================================================

mcp = FastMCP(
    name="private_knowledge_server",
    stateless_http=True,
    json_response=True
)


@mcp.tool()
def search_private_knowledge(query: str) -> str:
    query = query.lower()

    if any(word in query for word in ["owner", "ownership", "failure", "failures"]):
        return "The fallback owner for Blue Mango failures is Aniket from the DS Ops team."

    if any(word in query for word in ["0.62", "confidence", "threshold"]):
        return "If Blue Mango confidence drops below 0.62, the audience must be routed to the Silver Gate manual review queue."

    if "silver gate" in query:
        return "Silver Gate is the manual review queue for Project Sarayu."

    if "blue mango" in query:
        return "Blue Mango is the internal codename for the customer retention model in Project Sarayu."

    return "No private knowledge found."


def run_mcp_server():
    mcp.settings.port = PORT
    mcp.run(transport="streamable-http")


def port_is_busy(port):
    with socket.socket(socket.AF_INET, socket.SOCK_STREAM) as s:
        return s.connect_ex(("127.0.0.1", port)) == 0


if not port_is_busy(PORT):
    server_thread = threading.Thread(
        target=run_mcp_server,
        daemon=True
    )
    server_thread.start()
    time.sleep(3)
    print(f"MCP server started on port {PORT}.")
else:
    print(f"MCP server already running on port {PORT}. Reusing it.")


# ============================================================
# MCP CLIENT
# ============================================================

class MCPClient:
    def __init__(self, url=MCP_URL):
        self.url = url

    async def call_tool(self, tool_name, arguments):
        async with streamablehttp_client(self.url) as (read, write, _):
            async with ClientSession(read, write) as session:
                await session.initialize()

                result = await session.call_tool(
                    name=tool_name,
                    arguments=arguments
                )

                return result.content[0].text


# ============================================================
# GEMINI BASE AGENT USING types.Tool
# ============================================================

class BaseAgent:
    def __init__(self, name, sys_prompt, mcp_client, model="gemini-3.1-flash-lite"):
        self.name = name
        self.sys_prompt = sys_prompt
        self.mcp_client = mcp_client
        self.client = genai.Client(api_key=os.environ.get("GEMINI_API_KEY"))
        self.model = model
        self.memory = []

    async def think(self, input_text):
        tool = types.Tool(
            function_declarations=[
                types.FunctionDeclaration(
                    name="search_private_knowledge",
                    description="Search private MCP knowledge base for internal project facts.",
                    parameters=types.Schema(
                        type=types.Type.OBJECT,
                        properties={
                            "query": types.Schema(
                                type=types.Type.STRING,
                                description="Search query for private knowledge"
                            )
                        },
                        required=["query"]
                    )
                )
            ]
        )

        response = self.client.models.generate_content(
            model=self.model,
            contents=[
                types.Content(
                    role="user",
                    parts=[types.Part.from_text(text=input_text)]
                )
            ],
            config=types.GenerateContentConfig(
                thinking_config=types.ThinkingConfig(thinking_budget=0),
                system_instruction=[types.Part.from_text(text=self.sys_prompt)],
                tools=[tool]
            )
        )

        part = response.candidates[0].content.parts[0]

        if part.function_call:
            tool_name = part.function_call.name
            arguments = dict(part.function_call.args)

            print(f"\n[Gemini selected MCP tool: {tool_name}]")
            print(arguments)

            answer = await self.mcp_client.call_tool(
                tool_name=tool_name,
                arguments=arguments
            )

        else:
            answer = response.text

        print(answer)
        self.memory.append((input_text, answer))
        return answer


# ============================================================
# MCP RAG AGENT
# ============================================================

class MCPRAGAgent(BaseAgent):
    def __init__(self, name, mcp_client):
        super().__init__(
            name=name,
            sys_prompt="""
You are an MCP RAG Agent.

Use the MCP tool when the user asks about private/internal project facts.
Do not invent private information.
""",
            mcp_client=mcp_client
        )


# ============================================================
# ORCHESTRATION
# ============================================================

async def main():
    mcp_client = MCPClient()

    agent = MCPRAGAgent(
        name="MCPRAGAgent",
        mcp_client=mcp_client
    )

    questions = [
        "What is Blue Mango?",
        "Who owns Blue Mango failures?",
        "What happens if confidence drops below 0.62?",
        "What is Silver Gate?"
    ]

    for question in questions:
        print(f"\nUser: {question}")
        print(f"{agent.name}: ", end="")
        await agent.think(question)


await main()
Mental Model

Lines 1-18: Install dependencies, import Gemini and MCP components, patch the notebook event loop with nest_asyncio, and set a placeholder API key.

Lines 25-26: Define the MCP server port and URL. The server will expose tools at /mcp.

Lines 33-59: Create a real FastMCP server and register search_private_knowledge as a tool. This tool acts like a tiny private knowledge service.

Lines 62-82: Start the MCP server only if the port is free. If the server is already running, the script reuses it instead of crashing.

Lines 89-104: MCPClient connects to the MCP server over streamable HTTP, initializes a session, calls a tool by name, and returns the text result.

Lines 111-183: BaseAgent declares the MCP tool to Gemini using types.Tool. If Gemini returns a function call, Python executes that call through the MCP client.

Lines 190-204: MCPRAGAgent specializes the base agent with a system prompt: use MCP for private facts and do not invent internal information.

Lines 211-232: The notebook orchestration creates the client, creates the agent, asks four private-knowledge questions, and awaits each response.
